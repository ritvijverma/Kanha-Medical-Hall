  c.width=innerWidth;c.height=innerHeight;
  const conf=Array.from({length:150},()=>({
    x:Math.random()*c.width,
    y:Math.random()*c.height-c.height,
    r:Math.random()*6+4,
    d:Math.random()*10,

    ctx.clearRect(0,0,c.width,c.height);
    conf.forEach(p=>{
      ctx.fillStyle=p.col;
      ctx.beginPath();
      ctx.arc(p.x,p.y,p.r,0,Math.PI*2);

    });
